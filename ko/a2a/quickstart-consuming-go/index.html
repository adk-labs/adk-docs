    ADKм—җм„ң м§ҖмӣҗGoмӢӨн—ҳм Ғ кё°лҠҘ

    мқҙ л№ лҘё мӢңмһ‘мқҖ лӘЁл“  к°ңл°ңмһҗм—җкІҢ к°ҖмһҘ мқјл°ҳм Ғмқё мӢңмһ‘ м§Җм җмқё "мӣҗкІ© м—җмқҙм „нҠёк°Җ мһҲлҠ”лҚ°, лӮҙ ADK м—җмқҙм „нҠёк°Җ A2AлҘј нҶөн•ҙ мқҙлҘј мӮ¬мҡ©н•ҳкІҢ н•ҳл Өл©ҙ м–ҙл–»кІҢ н•ҙм•ј н•ҳлӮҳмҡ”?"м—җ лҢҖн•ҙ лӢӨлЈ№лӢҲлӢӨ. мқҙлҠ” м„ңлЎң лӢӨлҘё м—җмқҙм „нҠёк°Җ нҳ‘л Ҙн•ҳкі  мғҒнҳё мһ‘мҡ©н•ҙм•ј н•ҳлҠ” ліөмһЎн•ң лӢӨмӨ‘ м—җмқҙм „нҠё мӢңмҠӨн…ңмқ„ кө¬м¶•н•ҳлҠ” лҚ° мӨ‘мҡ”н•©лӢҲлӢӨ.

    к°ңмҡ”¶

    мқҙ мғҳн”ҢмқҖ м—җмқҙм „нҠё к°ңл°ң нӮӨнҠё(ADK)мқҳ м—җмқҙм „нҠё лҢҖ м—җмқҙм „нҠё(A2A) м•„нӮӨн…ҚмІҳлҘј мӢңм—°н•ҳл©°, м—¬лҹ¬ м—җмқҙм „нҠёк°Җ н•Ёк»ҳ мһ‘лҸҷн•ҳм—¬ ліөмһЎн•ң мһ‘м—…мқ„ мІҳлҰ¬н•ҳлҠ” л°©лІ•мқ„ ліҙм—¬мӨҚлӢҲлӢӨ. мқҙ мғҳн”ҢмқҖ мЈјмӮ¬мң„лҘј көҙлҰ¬кі  мҲ«мһҗк°Җ мҶҢмҲҳмқём§Җ нҷ•мқён•  мҲҳ мһҲлҠ” м—җмқҙм „нҠёлҘј кө¬нҳ„н•©лӢҲлӢӨ.

    a2a_basic/main.go
    func newPrimeAgent() (agent.Agent, error) {
    remoteAgent, err := remoteagent.NewA2A(remoteagent.A2AConfig{
        Name:            "prime_agent",
        Description:     "Agent that handles checking if numbers are prime.",
        AgentCardSource: "http://localhost:8001",

    if err != nil {

}

    к·ёлҹ° лӢӨмқҢ лЈЁнҠё м—җмқҙм „нҠём—җм„ң мӣҗкІ© м—җмқҙм „нҠёлҘј к°„лӢЁнһҲ мӮ¬мҡ©н•  мҲҳ мһҲмҠөлӢҲлӢӨ. мқҙ кІҪмҡ° primeAgentлҠ” м•„лһҳ root_agentмқҳ н•ҳмң„ м—җмқҙм „нҠё мӨ‘ н•ҳлӮҳлЎң мӮ¬мҡ©лҗ©лӢҲлӢӨ.

    a2a_basic/main.go
    func newRootAgent(ctx context.Context, rollAgent, primeAgent agent.Agent) (agent.Agent, error) {
    model, err := gemini.NewModel(ctx, "gemini-2.0-flash", &genai.ClientConfig{})
    if err != nil {
        return nil, err
    }
    return llmagent.New(llmagent.Config{
        Name:  "root_agent",
        Model: model,
        Instruction: `
      You are a helpful assistant that can roll dice and check if numbers are prime.
      You delegate rolling dice tasks to the roll_agent and prime checking tasks to the prime_agent.
      Follow these steps:
      1. If the user asks to roll a die, delegate to the roll_agent.
      2. If the user asks to check primes, delegate to the prime_agent.
      3. If the user asks to roll a die and then check if the result is prime, call roll_agent first, then pass the result to prime_agent.
      Always clarify the results before proceeding.
    `,
        SubAgents: []agent.Agent{rollAgent, primeAgent},
        Tools:     []tool.Tool{},
    })
}

    мғҒнҳё мһ‘мҡ© мҳҲ¶

    кё°ліё м—җмқҙм „нҠёмҷҖ мӣҗкІ© м—җмқҙм „нҠёк°Җ лӘЁл‘җ мӢӨн–үлҗҳл©ҙ лЈЁнҠё м—җмқҙм „нҠёмҷҖ мғҒнҳё мһ‘мҡ©н•ҳм—¬ A2AлҘј нҶөн•ҙ мӣҗкІ© м—җмқҙм „нҠёлҘј нҳём¶ңн•ҳлҠ” л°©лІ•мқ„ нҷ•мқён•  мҲҳ мһҲмҠөлӢҲлӢӨ.

    к°„лӢЁн•ң мЈјмӮ¬мң„ көҙлҰ¬кё°: мқҙ мғҒнҳё мһ‘мҡ©мқҖ лЎңм»¬ м—җмқҙм „нҠёмқё лЎӨ м—җмқҙм „нҠёлҘј мӮ¬мҡ©н•©лӢҲлӢӨ.

    мӮ¬мҡ©мһҗ: 6л©ҙмІҙ мЈјмӮ¬мң„лҘј көҙл ӨмЈјм„ёмҡ”
лҙҮмқҙ лҸ„кө¬ нҳём¶ң: transfer_to_agent, мқёмҲҳ: map[agent_name:roll_agent]
лҙҮмқҙ лҸ„кө¬ нҳём¶ң: roll_die, мқёмҲҳ: map[sides:6]
лҙҮ: 6л©ҙмІҙ мЈјмӮ¬мң„лҘј көҙл ёкі  кІ°кіјлҠ” 6мһ…лӢҲлӢӨ.

    мҶҢмҲҳ нҷ•мқё:

    мқҙ мғҒнҳё мһ‘мҡ©мқҖ A2AлҘј нҶөн•ҙ мӣҗкІ© м—җмқҙм „нҠёмқё н”„лқјмһ„ м—җмқҙм „нҠёлҘј мӮ¬мҡ©н•©лӢҲлӢӨ.

    мӮ¬мҡ©мһҗ: 7мқҖ мҶҢмҲҳмқёк°Җмҡ”?
лҙҮмқҙ лҸ„кө¬ нҳём¶ң: transfer_to_agent, мқёмҲҳ: map[agent_name:prime_agent]
лҙҮмқҙ лҸ„кө¬ нҳём¶ң: prime_checking, мқёмҲҳ: map[nums:[7]]
лҙҮ: мҳҲ, 7мқҖ мҶҢмҲҳмһ…лӢҲлӢӨ.

    кІ°н•©лҗң мһ‘м—…:

    мқҙ мғҒнҳё мһ‘мҡ©мқҖ лЎңм»¬ лЎӨ м—җмқҙм „нҠёмҷҖ мӣҗкІ© н”„лқјмһ„ м—җмқҙм „нҠёлҘј лӘЁл‘җ мӮ¬мҡ©н•©лӢҲлӢӨ.

    мӮ¬мҡ©мһҗ: мЈјмӮ¬мң„лҘј көҙл Өм„ң мҶҢмҲҳмқём§Җ нҷ•мқён•ҙмЈјм„ёмҡ”
лҙҮ: м•ҢкІ мҠөлӢҲлӢӨ. лЁјм Җ мЈјмӮ¬мң„лҘј көҙлҰ° лӢӨмқҢ кІ°кіјк°Җ мҶҢмҲҳмқём§Җ нҷ•мқён•ҳкІ мҠөлӢҲлӢӨ.

лҙҮмқҙ лҸ„кө¬ нҳём¶ң: transfer_to_agent, мқёмҲҳ: map[agent_name:roll_agent]
лҙҮмқҙ лҸ„кө¬ нҳём¶ң: roll_die, мқёмҲҳ: map[sides:6]
лҙҮмқҙ лҸ„кө¬ нҳём¶ң: transfer_to_agent, мқёмҲҳ: map[agent_name:prime_agent]
лҙҮмқҙ лҸ„кө¬ нҳём¶ң: prime_checking, мқёмҲҳ: map[nums:[3]]
лҙҮ: 3мқҖ мҶҢмҲҳмһ…лӢҲлӢӨ.

    лӢӨмқҢ лӢЁкі„¶

    мқҙм ң A2A м„ңлІ„лҘј нҶөн•ҙ мӣҗкІ© м—җмқҙм „нҠёлҘј мӮ¬мҡ©н•ҳлҠ” м—җмқҙм „нҠёлҘј л§Ңл“Өм—ҲмңјлҜҖлЎң лӢӨмқҢ лӢЁкі„лҠ” мһҗмӢ мқҳ м—җмқҙм „нҠёлҘј л…ём¶ңн•ҳлҠ” л°©лІ•мқ„ л°°мҡ°лҠ” кІғмһ…лӢҲлӢӨ.
