    лҹ°нғҖмһ„¶

    ADKм—җм„ң м§ҖмӣҗPython v0.1.0Go v0.1.0Java v0.1.0

    ADK лҹ°нғҖмһ„мқҖ мӮ¬мҡ©мһҗ мғҒнҳё мһ‘мҡ© мӨ‘м—җ м—җмқҙм „нҠё м• н”ҢлҰ¬мјҖмқҙм…ҳмқ„ кө¬лҸҷн•ҳлҠ” кё°ліё м—”м§„мһ…лӢҲлӢӨ. м •мқҳлҗң м—җмқҙм „нҠё, лҸ„кө¬ л°Ҹ мҪңл°ұмқ„ к°Җм ёмҷҖ мӮ¬мҡ©мһҗ мһ…л Ҙм—җ лҢҖн•ң мқ‘лӢөмңјлЎң мӢӨн–үмқ„ мҳӨмјҖмҠӨнҠёл Ҳмқҙм…ҳн•ҳкі  м •ліҙ нқҗлҰ„, мғҒнғң ліҖкІҪ л°Ҹ LLM лҳҗлҠ” мҠӨнҶ лҰ¬м§ҖмҷҖ к°ҷмқҖ мҷёл¶Җ м„ңл№„мҠӨмҷҖмқҳ мғҒнҳё мһ‘мҡ©мқ„ кҙҖлҰ¬н•ҳлҠ” мӢңмҠӨн…ңмһ…лӢҲлӢӨ.

    лҹ°нғҖмһ„мқ„ м—җмқҙм „нҠё м• н”ҢлҰ¬мјҖмқҙм…ҳмқҳ "м—”м§„"мңјлЎң мғқк°Ғн•ҳмӢӯмӢңмҳӨ. л¶Җн’Ҳ(м—җмқҙм „нҠё, лҸ„кө¬)мқ„ м •мқҳн•ҳл©ҙ лҹ°нғҖмһ„мқҖ мӮ¬мҡ©мһҗ мҡ”мІӯмқ„ мқҙн–үн•ҳкё° мң„н•ҙ л¶Җн’Ҳмқҙ м–ҙл–»кІҢ м—°кІ°лҗҳкі  н•Ёк»ҳ мӢӨн–үлҗҳлҠ”м§ҖлҘј мІҳлҰ¬н•©лӢҲлӢӨ.

    н•өмӢ¬ м•„мқҙл””м–ҙ: мқҙлІӨнҠё лЈЁн”„¶

    н•өмӢ¬м ҒмңјлЎң ADK лҹ°нғҖмһ„мқҖ мқҙлІӨнҠё лЈЁн”„м—җм„ң мһ‘лҸҷн•©лӢҲлӢӨ. мқҙ лЈЁн”„лҠ” Runner кө¬м„ұ мҡ”мҶҢмҷҖ м •мқҳлҗң "мӢӨн–ү л…јлҰ¬"(м—җмқҙм „нҠё, м—җмқҙм „нҠёк°Җ л§Ңл“ңлҠ” LLM нҳём¶ң, мҪңл°ұ л°Ҹ лҸ„кө¬ нҸ¬н•Ё) к°„мқҳ м–‘л°©н–Ҙ нҶөмӢ мқ„ мҡ©мқҙн•ҳкІҢ н•©лӢҲлӢӨ.

    к°„лӢЁнһҲ л§җн•ҙм„ң:

    1. RunnerлҠ” мӮ¬мҡ©мһҗ мҝјлҰ¬лҘј мҲҳмӢ н•ҳкі  кё°ліё Agentм—җ мІҳлҰ¬лҘј мӢңмһ‘н•ҳлҸ„лЎқ мҡ”мІӯн•©лӢҲлӢӨ.
    2. Agent(л°Ҹ кҙҖл Ё л…јлҰ¬)лҠ” ліҙкі н•  лӮҙмҡ©(мқ‘лӢө, лҸ„кө¬ мӮ¬мҡ© мҡ”мІӯ лҳҗлҠ” мғҒнғң ліҖкІҪ л“ұ)мқҙ мһҲмқ„ л•Ңк№Ңм§Җ мӢӨн–үлҗң лӢӨмқҢ EventлҘј мғқм„ұн•ҳкұ°лӮҳ л°©м¶ңн•©лӢҲлӢӨ.
    3. RunnerлҠ” мқҙ EventлҘј мҲҳмӢ н•ҳкі  кҙҖл Ё мһ‘м—…(мҳҲ: ServicesлҘј нҶөн•ң мғҒнғң ліҖкІҪ м ҖмһҘ)мқ„ мІҳлҰ¬н•ҳкі  мқҙлІӨнҠёлҘј м „лӢ¬н•©лӢҲлӢӨ(мҳҲ: мӮ¬мҡ©мһҗ мқён„°нҺҳмқҙмҠӨлЎң).
    4. Runnerк°Җ мқҙлІӨнҠёлҘј мІҳлҰ¬н•ң нӣ„м—җл§Ң Agentмқҳ л…јлҰ¬к°Җ мқјмӢң мӨ‘м§Җлҗң мң„м№ҳм—җм„ң мһ¬к°ңлҗҳл©°, мқҙм ң Runnerк°Җ м»Өл°Ӣн•ң ліҖкІҪ мӮ¬н•ӯмқҳ нҡЁкіјлҘј ліј мҲҳ мһҲмҠөлӢҲлӢӨ.
    5. мқҙ мЈјкё°лҠ” нҳ„мһ¬ мӮ¬мҡ©мһҗ мҝјлҰ¬м—җ лҢҖн•ҙ м—җмқҙм „нҠёк°Җ мғқм„ұн•  мқҙлІӨнҠёк°Җ лҚ” мқҙмғҒ м—Ҷмқ„ л•Ңк№Ңм§Җ л°ҳліөлҗ©лӢҲлӢӨ.

    мқҙ мқҙлІӨнҠё кё°л°ҳ лЈЁн”„лҠ” ADKк°Җ м—җмқҙм „нҠё мҪ”л“ңлҘј мӢӨн–үн•ҳлҠ” л°©лІ•мқ„ м ңм–ҙн•ҳлҠ” кё°ліё нҢЁн„ҙмһ…лӢҲлӢӨ.

    мӢ¬мһҘ л°•лҸҷ: мқҙлІӨнҠё лЈЁн”„ - лӮҙл¶Җ мһ‘лҸҷ¶

    мқҙлІӨнҠё лЈЁн”„лҠ” RunnerмҷҖ мӮ¬мҡ©мһҗ м§Җм • мҪ”л“ң(м—җмқҙм „нҠё, лҸ„кө¬, мҪңл°ұ, м„Өкі„ л¬ём„ңм—җм„ң "мӢӨн–ү л…јлҰ¬" лҳҗлҠ” "л…јлҰ¬ кө¬м„ұ мҡ”мҶҢ"лқјкі  нҶөм№ӯ) к°„мқҳ мғҒнҳё мһ‘мҡ©мқ„ м •мқҳн•ҳлҠ” н•өмӢ¬ мҡҙмҳҒ нҢЁн„ҙмһ…лӢҲлӢӨ. мқҙлҠ” мұ…мһ„мқҳ лӘ…нҷ•н•ң л¶„н• мқ„ м„Өм •н•©лӢҲлӢӨ.

    Note

    нҠ№м • л©”м„ңл“ң мқҙлҰ„ л°Ҹ л§Өк°ңліҖмҲҳ мқҙлҰ„мқҖ SDK м–ём–ҙм—җ л”°лқј м•Ҫк°„ лӢӨлҘј мҲҳ мһҲмҠөлӢҲлӢӨ(мҳҲ: Pythonмқҳ agent_to_run.run_async(...), Goмқҳ agent.Run(...), Javaмқҳ agent_to_run.runAsync(...)). мһҗм„ён•ң лӮҙмҡ©мқҖ м–ём–ҙлі„ API м„ӨлӘ…м„ңлҘј м°ёмЎ°н•ҳмӢӯмӢңмҳӨ.

    Runnerмқҳ м—ӯн• (мҳӨмјҖмҠӨнҠёл Ҳмқҙн„°)¶

    RunnerлҠ” лӢЁмқј мӮ¬мҡ©мһҗ нҳём¶ңм—җ лҢҖн•ң мӨ‘м•ҷ мЎ°м •мһҗ м—ӯн• мқ„ н•©лӢҲлӢӨ. лЈЁн”„м—җм„ңмқҳ мұ…мһ„мқҖ лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

    1. мӢңмһ‘: мөңмў… мӮ¬мҡ©мһҗмқҳ мҝјлҰ¬(new_message)лҘј мҲҳмӢ н•ҳкі  мқјл°ҳм ҒмңјлЎң SessionServiceлҘј нҶөн•ҙ м„ём…ҳ кё°лЎқм—җ м¶”к°Җн•©лӢҲлӢӨ.
    2. мӢңмһ‘: кё°ліё м—җмқҙм „нҠёмқҳ мӢӨн–ү л©”м„ңл“ң(мҳҲ: agent_to_run.run_async(...))лҘј нҳём¶ңн•ҳм—¬ мқҙлІӨнҠё мғқм„ұ н”„лЎңм„ёмҠӨлҘј мӢңмһ‘н•©лӢҲлӢӨ.
    3. мҲҳмӢ  л°Ҹ мІҳлҰ¬: м—җмқҙм „нҠё л…јлҰ¬к°Җ EventлҘј мғқм„ұн•ҳкұ°лӮҳ л°©м¶ңн•  л•Ңк№Ңм§Җ кё°лӢӨлҰҪлӢҲлӢӨ. мқҙлІӨнҠёлҘј мҲҳмӢ н•ҳл©ҙ RunnerлҠ” мҰүмӢң мІҳлҰ¬н•©лӢҲлӢӨ. м—¬кё°м—җлҠ” лӢӨмқҢмқҙ нҸ¬н•Ёлҗ©лӢҲлӢӨ.

      • кё°нғҖ лӮҙл¶Җ л¶Җкё° мһ‘м—…мқ„ мҲҳн–үн•©лӢҲлӢӨ.
    4. м—…мҠӨнҠёлҰј мғқм„ұ: мІҳлҰ¬лҗң мқҙлІӨнҠёлҘј м „лӢ¬н•©лӢҲлӢӨ(мҳҲ: л ҢлҚ”л§Ғмқ„ мң„н•ҙ нҳём¶ң м• н”ҢлҰ¬мјҖмқҙм…ҳ лҳҗлҠ” UIлЎң).
    5. л°ҳліө: мғқм„ұлҗң мқҙлІӨнҠём—җ лҢҖн•ң мІҳлҰ¬к°Җ мҷ„лЈҢлҗҳм—ҲмқҢмқ„ м—җмқҙм „нҠё л…јлҰ¬м—җ мӢ нҳёлҘј ліҙлӮҙ лӢӨмқҢ мқҙлІӨнҠёлҘј мһ¬к°ңн•ҳкі  мғқм„ұн•  мҲҳ мһҲлҸ„лЎқ н•©лӢҲлӢӨ.

        # 4. м—…мҠӨнҠёлҰј мІҳлҰ¬лҘј мң„н•ҙ мқҙлІӨнҠё мғқм„ұ(мҳҲ: UI л ҢлҚ”л§Ғ)
        yield event
        # RunnerлҠ” м—җмқҙм „нҠё мғқм„ұкё°к°Җ мғқм„ұ нӣ„ кі„мҶҚлҗ  мҲҳ мһҲмқҢмқ„ м•”мӢңм ҒмңјлЎң мӢ нҳёлҘј ліҙлғ…лӢҲлӢӨ.

func (r *Runner) RunConceptual(ctx context.Context, session *session.Session, newQuery *genai.Content) iter.Seq2[*Event, error] {
    return func(yield func(*Event, error) bool) {
        # 1. new_queryлҘј м„ём…ҳ мқҙлІӨнҠё кё°лЎқм—җ м¶”к°Җ(SessionServiceлҘј нҶөн•ҙ)
        # ...
        userEvent := session.NewEvent(ctx.InvocationID()) # к°ңл…җм Ғ ліҙкё°лҘј мң„н•ҙ лӢЁмҲңнҷ”
        userEvent.Author = "user"
        userEvent.LLMResponse = model.LLMResponse{Content: newQuery}

        if _, err := r.sessionService.Append(ctx, &session.AppendRequest{Event: userEvent}); err != nil {
            yield(nil, err)
            return
        }

        # 2. м—җмқҙм „нҠёлҘј нҳём¶ңн•ҳм—¬ мқҙлІӨнҠё мҠӨнҠёлҰј мӢңмһ‘
        # agent.RunлҸ„ iter.Seq2[*Event, error]лҘј л°ҳнҷҳн•ңлӢӨкі  к°Җм •
        agentEventsAndErrs := r.agent.Run(ctx, &agent.RunRequest{Session: session, Input: newQuery})

        for event, err := range agentEventsAndErrs {
            if err != nil {
                if !yield(event, err) { # мҳӨлҘҳк°Җ мһҲлҚ”лқјлҸ„ мқҙлІӨнҠёлҘј мғқм„ұн•ң лӢӨмқҢ мӨ‘м§Җ
                    return
                }
                return # м—җмқҙм „нҠёк°Җ мҳӨлҘҳлЎң мҷ„лЈҢлҗЁ
            }

            # 3. мғқм„ұлҗң мқҙлІӨнҠёлҘј мІҳлҰ¬н•ҳкі  ліҖкІҪ мӮ¬н•ӯ м»Өл°Ӣ
            # л¶Җл¶„м Ғмқҙм§Җ м•ҠмқҖ мқҙлІӨнҠёл§Ң м„ём…ҳ м„ңл№„мҠӨм—җ м»Өл°Ӣ(мӢӨм ң мҪ”л“ңм—җм„ң ліј мҲҳ мһҲл“Ҝмқҙ)
            if !event.LLMResponse.Partial {
                if _, err := r.sessionService.Append(ctx, &session.AppendRequest{Event: event}); err != nil {
                    yield(nil, err)
                    return
                }
            }
            # memory_service.update_memory(...) # н•ҙлӢ№лҗҳлҠ” кІҪмҡ°
            # artifact_serviceлҠ” м—җмқҙм „нҠё мӢӨн–ү мӨ‘м—җ м»Ён…ҚмҠӨнҠёлҘј нҶөн•ҙ мқҙлҜё нҳём¶ңлҗҳм—Ҳмқ„ мҲҳ мһҲмҠөлӢҲлӢӨ.

            # 4. м—…мҠӨнҠёлҰј мІҳлҰ¬лҘј мң„н•ҙ мқҙлІӨнҠё мғқм„ұ
            if !yield(event, nil) {
                return # м—…мҠӨнҠёлҰј мҶҢл№„мһҗк°Җ мӨ‘м§ҖлҗЁ
            }
        }
        # м—җмқҙм „нҠёк°Җ м„ұкіөм ҒмңјлЎң мҷ„лЈҢлҗЁ
    }
}

    1. Runner¶

      • м—ӯн• : лӢЁмқј мӮ¬мҡ©мһҗ мҝјлҰ¬(run_async)м—җ лҢҖн•ң кё°ліё м§„мһ…м җ л°Ҹ мҳӨмјҖмҠӨнҠёл Ҳмқҙн„°мһ…лӢҲлӢӨ.
      • кё°лҠҘ: м „мІҙ мқҙлІӨнҠё лЈЁн”„лҘј кҙҖлҰ¬н•ҳкі , мӢӨн–ү л…јлҰ¬м—җм„ң мғқм„ұлҗң мқҙлІӨнҠёлҘј мҲҳмӢ н•ҳкі , м„ңл№„мҠӨмҷҖ нҳ‘л Ҙн•ҳм—¬ мқҙлІӨнҠё мһ‘м—…(мғҒнғң/м•„нӢ°нҢ©нҠё ліҖкІҪ)мқ„ мІҳлҰ¬ л°Ҹ м»Өл°Ӣн•ҳкі , мІҳлҰ¬лҗң мқҙлІӨнҠёлҘј м—…мҠӨнҠёлҰј(мҳҲ: UI)мңјлЎң м „лӢ¬н•©лӢҲлӢӨ. кё°ліём ҒмңјлЎң мғқм„ұлҗң мқҙлІӨнҠёлҘј кё°л°ҳмңјлЎң лҢҖнҷ”лҘј м°ЁлЎҖлЎң м§„н–үн•©лӢҲлӢӨ. (google.adk.runners.runnerм—җ м •мқҳлҗЁ).

    2. мӢӨн–ү л…јлҰ¬ кө¬м„ұ мҡ”мҶҢ¶

      • м—ӯн• : мӮ¬мҡ©мһҗ м§Җм • мҪ”л“ңмҷҖ н•өмӢ¬ м—җмқҙм „нҠё кё°лҠҘмқҙ нҸ¬н•Ёлҗң л¶Җл¶„мһ…лӢҲлӢӨ.
      • кө¬м„ұ мҡ”мҶҢ:
      • Agent(BaseAgent, LlmAgent л“ұ): м •ліҙлҘј мІҳлҰ¬н•ҳкі  мһ‘м—…мқ„ кІ°м •н•ҳлҠ” кё°ліё л…јлҰ¬ лӢЁмң„мһ…лӢҲлӢӨ. мқҙлІӨнҠёлҘј мғқм„ұн•ҳлҠ” _run_async_impl л©”м„ңл“ңлҘј кө¬нҳ„н•©лӢҲлӢӨ.
      • Tools(BaseTool, FunctionTool, AgentTool л“ұ): м—җмқҙм „нҠё(мў…мў… LlmAgent)к°Җ мҷёл¶Җ м„ёкі„мҷҖ мғҒнҳё мһ‘мҡ©н•ҳкұ°лӮҳ нҠ№м • мһ‘м—…мқ„ мҲҳн–үн•ҳлҠ” лҚ° мӮ¬мҡ©н•ҳлҠ” мҷёл¶Җ н•ЁмҲҳ лҳҗлҠ” кё°лҠҘмһ…лӢҲлӢӨ. мӢӨн–үн•ҳкі  кІ°кіјлҘј л°ҳнҷҳн•ҳл©°, мқҙ кІ°кіјлҠ” мқҙлІӨнҠёлЎң лһҳн•‘лҗ©лӢҲлӢӨ.
      • Callbacks(н•ЁмҲҳ): м—җмқҙм „нҠём—җ м—°кІ°лҗң мӮ¬мҡ©мһҗ м •мқҳ н•ЁмҲҳ(мҳҲ: before_agent_callback, after_model_callback)лЎң, мӢӨн–ү нқҗлҰ„мқҳ нҠ№м • м§Җм җм—җ м—°кІ°лҗҳм–ҙ мһ мһ¬м ҒмңјлЎң лҸҷмһ‘мқҙлӮҳ мғҒнғңлҘј мҲҳм •н•ҳл©°, к·ё нҡЁкіјлҠ” мқҙлІӨнҠём—җ мәЎмІҳлҗ©лӢҲлӢӨ.
      • кё°лҠҘ: мӢӨм ң мӮ¬кі , кі„мӮ° лҳҗлҠ” мҷёл¶Җ мғҒнҳё мһ‘мҡ©мқ„ мҲҳн–үн•©лӢҲлӢӨ. Event к°қмІҙлҘј мғқм„ұн•ҳкі  Runnerк°Җ мІҳлҰ¬н•  л•Ңк№Ңм§Җ мқјмӢң мӨ‘м§Җн•ҳм—¬ кІ°кіјлӮҳ мҡ”кө¬ мӮ¬н•ӯмқ„ м „лӢ¬н•©лӢҲлӢӨ.

    3. Event¶

      • м—ӯн• : RunnerмҷҖ мӢӨн–ү л…јлҰ¬ к°„м—җ мЈјкі л°ӣлҠ” л©”мӢңм§Җмһ…лӢҲлӢӨ.
      • кё°лҠҘ: мӣҗмһҗм Ғ л°ңмғқ(мӮ¬мҡ©мһҗ мһ…л Ҙ, м—җмқҙм „нҠё н…ҚмҠӨнҠё, лҸ„кө¬ нҳём¶ң/кІ°кіј, мғҒнғң ліҖкІҪ мҡ”мІӯ, м ңм–ҙ мӢ нҳё)мқ„ лӮҳнғҖлғ…лӢҲлӢӨ. л°ңмғқ лӮҙмҡ©кіј мқҳлҸ„лҗң л¶Җмһ‘мҡ©(state_deltaмҷҖ к°ҷмқҖ actions)мқ„ лӘЁл‘җ м „лӢ¬н•©лӢҲлӢӨ.

      • кө¬м„ұ мҡ”мҶҢ:
      • SessionService(BaseSessionService, InMemorySessionService л“ұ): Session к°қмІҙлҘј кҙҖлҰ¬н•ҳл©°, м ҖмһҘ/лЎңл“ң, м„ём…ҳ мғҒнғңм—җ state_delta м Ғмҡ©, event historyм—җ мқҙлІӨнҠё м¶”к°Җ л“ұмқ„ нҸ¬н•Ён•©лӢҲлӢӨ.
      • ArtifactService(BaseArtifactService, InMemoryArtifactService, GcsArtifactService л“ұ): мқҙм§„ м•„нӢ°нҢ©нҠё лҚ°мқҙн„°мқҳ м ҖмһҘ л°Ҹ кІҖмғүмқ„ кҙҖлҰ¬н•©лӢҲлӢӨ. мӢӨн–ү л…јлҰ¬ мӨ‘м—җ м»Ён…ҚмҠӨнҠёлҘј нҶөн•ҙ save_artifactк°Җ нҳём¶ңлҗҳм§Җл§Ң мқҙлІӨнҠёмқҳ artifact_deltaлҠ” Runner/SessionServiceм—җ лҢҖн•ң мһ‘м—…мқ„ нҷ•мқён•©лӢҲлӢӨ.
      • MemoryService(BaseMemoryService л“ұ): (м„ нғқ мӮ¬н•ӯ) мӮ¬мҡ©мһҗм—җ лҢҖн•ң м„ём…ҳ к°„мқҳ мһҘкё° мқҳлҜё кё°м–өмқ„ кҙҖлҰ¬н•©лӢҲлӢӨ.
      • кё°лҠҘ: м§ҖмҶҚм„ұ кі„мёөмқ„ м ңкіөн•©лӢҲлӢӨ. RunnerлҠ” event.actionsм—җм„ң мӢ нҳёлҗң ліҖкІҪ мӮ¬н•ӯмқҙ мӢӨн–ү л…јлҰ¬к°Җ мһ¬к°ңлҗҳкё° м „м—җ м•Ҳм •м ҒмңјлЎң м ҖмһҘлҗҳлҸ„лЎқ н•ҳкё° мң„н•ҙ мқҙл“Өкіј мғҒнҳё мһ‘мҡ©н•©лӢҲлӢӨ.

    5. Session¶

      • кё°лҠҘ: нҳ„мһ¬ state мӮ¬м „, лӘЁл“  кіјкұ° events(event history) лӘ©лЎқ л°Ҹ кҙҖл Ё м•„нӢ°нҢ©нҠём—җ лҢҖн•ң м°ёмЎ°лҘј м ҖмһҘн•©лӢҲлӢӨ. SessionServiceм—җм„ң кҙҖлҰ¬н•ҳлҠ” мғҒнҳё мһ‘мҡ©мқҳ кё°ліё л ҲмҪ”л“ңмһ…лӢҲлӢӨ.

    мқҙлҹ¬н•ң н”Ңл Ҳмқҙм–ҙлҠ” мқҙлІӨнҠё лЈЁн”„лҘј нҶөн•ҙ м§ҖмҶҚм ҒмңјлЎң мғҒнҳё мһ‘мҡ©н•ҳм—¬ мӮ¬мҡ©мһҗ мҡ”мІӯмқ„ мІҳлҰ¬н•©лӢҲлӢӨ.

    мһ‘лҸҷ л°©мӢқ: лӢЁмҲңнҷ”лҗң нҳём¶ң¶

    LLM м—җмқҙм „нҠёк°Җ лҸ„кө¬лҘј нҳём¶ңн•ҳлҠ” мқјл°ҳм Ғмқё мӮ¬мҡ©мһҗ мҝјлҰ¬м—җ лҢҖн•ң лӢЁмҲңнҷ”лҗң нқҗлҰ„мқ„ м¶”м Ғн•ҙ ліҙкІ мҠөлӢҲлӢӨ.

    лӢЁкі„лі„ л¶„м„қ¶

    1. мӮ¬мҡ©мһҗ мһ…л Ҙ: мӮ¬мҡ©мһҗк°Җ мҝјлҰ¬лҘј ліҙлғ…лӢҲлӢӨ(мҳҲ: "н”„лһ‘мҠӨмқҳ мҲҳлҸ„лҠ” м–ҙл””мһ…лӢҲк№Ң?").

    3. м—җмқҙм „нҠё мӢӨн–ү: RunnerлҠ” м§Җм •лҗң лЈЁнҠё м—җмқҙм „нҠё(мҳҲ: LlmAgent)м—җм„ң agent.run_async(ctx)лҘј нҳём¶ңн•©лӢҲлӢӨ.
    4. LLM нҳём¶ң(мҳҲ): Agent_LlmмқҖ лҸ„кө¬лҘј нҳём¶ңн•ҳм—¬ м •ліҙк°Җ н•„мҡ”н•ҳлӢӨкі  нҢҗлӢЁн•©лӢҲлӢӨ. LLMм—җ лҢҖн•ң мҡ”мІӯмқ„ мӨҖл№„н•©лӢҲлӢӨ. LLMмқҙ MyToolмқ„ нҳём¶ңн•ҳкё°лЎң кІ°м •н–ҲлӢӨкі  к°Җм •н•©лӢҲлӢӨ.

    6. м—җмқҙм „нҠё мқјмӢң мӨ‘м§Җ: Agent_Llmмқҳ мӢӨн–үмқҖ мғқм„ұ м§Ғнӣ„м—җ мқјмӢң мӨ‘м§Җлҗ©лӢҲлӢӨ.

    8. м—җмқҙм „нҠё мһ¬к°ң: RunnerлҠ” мқҙлІӨнҠёк°Җ мІҳлҰ¬лҗҳм—ҲмқҢмқ„ мӢ нҳён•ҳкі  Agent_LlmмқҖ мӢӨн–үмқ„ мһ¬к°ңн•©лӢҲлӢӨ.

    10. лҸ„кө¬ кІ°кіј л°ҳнҷҳ: MyToolмқҙ мӢӨн–үлҗҳкі  кІ°кіјлҘј л°ҳнҷҳн•©лӢҲлӢӨ(мҳҲ: {'result': 'Paris'}).

    мҠӨнҠёлҰ¬л°Қ лҢҖ л№„мҠӨнҠёлҰ¬л°Қ м¶ңл Ҙ(partial=True)¶

    мқҙкІғмқҖ мЈјлЎң LLMмқҳ мқ‘лӢөмқҙ мІҳлҰ¬лҗҳлҠ” л°©мӢқкіј кҙҖл Ёмқҙ мһҲмңјл©°, нҠ№нһҲ мҠӨнҠёлҰ¬л°Қ мғқм„ұ APIлҘј мӮ¬мҡ©н•  л•Ң к·ёл ҮмҠөлӢҲлӢӨ.

    • мҠӨнҠёлҰ¬л°Қ: LLMмқҖ мқ‘лӢөмқ„ нҶ нҒ° лӢЁмң„ лҳҗлҠ” мһ‘мқҖ мІӯнҒ¬лЎң мғқм„ұн•©лӢҲлӢӨ.
    • н”„л Ҳмһ„мӣҢнҒ¬(мў…мў… BaseLlmFlow лӮҙ)лҠ” лӢЁмқј к°ңл…җм Ғ мқ‘лӢөм—җ лҢҖн•ҙ м—¬лҹ¬ Event к°қмІҙлҘј мғқм„ұн•©лӢҲлӢӨ. мқҙлҹ¬н•ң мқҙлІӨнҠёмқҳ лҢҖл¶Җл¶„мқҖ partial=TrueлҘј к°–мҠөлӢҲлӢӨ.
    • RunnerлҠ” partial=Trueмқё мқҙлІӨнҠёлҘј мҲҳмӢ н•ҳл©ҙ мқјл°ҳм ҒмңјлЎң мҰүмӢң м—…мҠӨнҠёлҰјмңјлЎң м „лӢ¬н•ҳм§Җл§Ң(UI н‘ңмӢңмҡ©) actions(state_delta л“ұ) мІҳлҰ¬лҠ” кұҙл„ҲлңҒлӢҲлӢӨ.
    • кІ°көӯ н”„л Ҳмһ„мӣҢнҒ¬лҠ” н•ҙлӢ№ мқ‘лӢөм—җ лҢҖн•ң мөңмў… мқҙлІӨнҠёлҘј мғқм„ұн•ҳл©°, мқҙлҠ” л¶Җл¶„м Ғмқҙм§Җ м•ҠмқҢ(partial=False лҳҗлҠ” turn_complete=TrueлҘј нҶөн•ҙ м•”мӢңм ҒмңјлЎң)мңјлЎң н‘ңмӢңлҗ©лӢҲлӢӨ.
    • RunnerлҠ” мқҙ мөңмў… мқҙлІӨнҠёл§Ң мҷ„м „нһҲ мІҳлҰ¬н•ҳм—¬ кҙҖл Ё state_delta лҳҗлҠ” artifact_deltaлҘј м»Өл°Ӣн•©лӢҲлӢӨ.
    • л№„мҠӨнҠёлҰ¬л°Қ: LLMмқҖ м „мІҙ мқ‘лӢөмқ„ н•ң лІҲм—җ мғқм„ұн•©лӢҲлӢӨ. н”„л Ҳмһ„мӣҢнҒ¬лҠ” л¶Җл¶„м Ғмқҙм§Җ м•ҠмқҢмңјлЎң н‘ңмӢңлҗң лӢЁмқј мқҙлІӨнҠёлҘј мғқм„ұн•ҳл©°, RunnerлҠ” мқҙлҘј мҷ„м „нһҲ мІҳлҰ¬н•©лӢҲлӢӨ.
    • мӨ‘мҡ”н•ң мқҙмң : н…ҚмҠӨнҠёк°Җ мғқм„ұлҗ  л•Ң UIк°Җ м җм§„м ҒмңјлЎң н‘ңмӢңлҗҳлҸ„лЎқ н—Ҳмҡ©н•ҳл©ҙм„ң LLMмқҳ мҷ„м „н•ң мқ‘лӢөмқ„ кё°л°ҳмңјлЎң мғҒнғң ліҖкІҪмқҙ мӣҗмһҗм ҒмңјлЎң н•ң лІҲл§Ң м Ғмҡ©лҗҳлҸ„лЎқ ліҙмһҘн•©лӢҲлӢӨ.

    л№„лҸҷкё°к°Җ кё°ліё(run_async)¶

    • н•өмӢ¬ м„Өкі„: ADK лҹ°нғҖмһ„мқҖ кё°ліём ҒмңјлЎң л№„лҸҷкё° лқјмқҙлёҢлҹ¬лҰ¬(Pythonмқҳ asyncio л°Ҹ Javaмқҳ RxJava л“ұ)лҘј кё°л°ҳмңјлЎң кө¬м¶•лҗҳм–ҙ лҸҷмӢң мһ‘м—…(LLM мқ‘лӢө лҳҗлҠ” лҸ„кө¬ мӢӨн–ү лҢҖкё° л“ұ)мқ„ м°ЁлӢЁн•ҳм§Җ м•Ҡкі  нҡЁмңЁм ҒмңјлЎң мІҳлҰ¬н•©лӢҲлӢӨ.
    • кё°ліё м§„мһ…м җ: Runner.run_asyncлҠ” м—җмқҙм „нҠё нҳём¶ңмқ„ мӢӨн–үн•ҳкё° мң„н•ң кё°ліё л©”м„ңл“ңмһ…лӢҲлӢӨ. лӘЁл“  н•өмӢ¬ мӢӨн–ү к°ҖлҠҘ кө¬м„ұ мҡ”мҶҢ(м—җмқҙм „нҠё, нҠ№м • нқҗлҰ„)лҠ” лӮҙл¶Җм ҒмңјлЎң л№„лҸҷкё° л©”м„ңл“ңлҘј мӮ¬мҡ©н•©лӢҲлӢӨ.
    • лҸҷкё° нҺёмқҳм„ұ(run): лҸҷкё° Runner.run л©”м„ңл“ңлҠ” мЈјлЎң нҺёмқҳлҘј мң„н•ҙ мЎҙмһ¬н•©лӢҲлӢӨ(мҳҲ: к°„лӢЁн•ң мҠӨнҒ¬лҰҪнҠё лҳҗлҠ” н…ҢмҠӨнҠё нҷҳкІҪ). к·ёлҹ¬лӮҳ лӮҙл¶Җм ҒмңјлЎң Runner.runмқҖ мқјл°ҳм ҒмңјлЎң Runner.run_asyncлҘј нҳём¶ңн•ҳкі  л№„лҸҷкё° мқҙлІӨнҠё лЈЁн”„ мӢӨн–үмқ„ кҙҖлҰ¬н•©лӢҲлӢӨ.
    • к°ңл°ңмһҗ кІҪн—ҳ: мөңмғҒмқҳ м„ұлҠҘмқ„ мң„н•ҙ м• н”ҢлҰ¬мјҖмқҙм…ҳ(ADKлҘј мӮ¬мҡ©н•ҳлҠ” мӣ№ м„ңлІ„ л“ұ)мқ„ л№„лҸҷкё°м ҒмңјлЎң м„Өкі„н•ҳлҠ” кІғмқҙ мўӢмҠөлӢҲлӢӨ. Pythonм—җм„ңлҠ” asyncioлҘј мӮ¬мҡ©н•ҳкі  Javaм—җм„ңлҠ” RxJavaмқҳ л°ҳмқ‘нҳ• н”„лЎңк·ёлһҳл°Қ лӘЁлҚёмқ„ нҷңмҡ©н•©лӢҲлӢӨ.
    • лҸҷкё° мҪңл°ұ/лҸ„кө¬: ADK н”„л Ҳмһ„мӣҢнҒ¬лҠ” лҸ„кө¬ л°Ҹ мҪңл°ұм—җ лҢҖн•ң л№„лҸҷкё° л°Ҹ лҸҷкё° н•ЁмҲҳлҘј лӘЁл‘җ м§Җмӣҗн•©лӢҲлӢӨ.
      • м°ЁлӢЁ I/O: мһҘкё° мӢӨн–ү лҸҷкё° I/O мһ‘м—…мқҳ кІҪмҡ° н”„л Ҳмһ„мӣҢнҒ¬лҠ” мӨ‘лӢЁмқ„ л°©м§Җн•ҳл Өкі  мӢңлҸ„н•©лӢҲлӢӨ. Python ADKлҠ” asyncio.to_threadлҘј мӮ¬мҡ©н•  мҲҳ мһҲмңјл©° Java ADKлҠ” мў…мў… м°ЁлӢЁ нҳём¶ңм—җ лҢҖн•ҙ м Ғм Ҳн•ң RxJava мҠӨмјҖмӨ„лҹ¬ лҳҗлҠ” лһҳнҚјм—җ мқҳмЎҙн•©лӢҲлӢӨ.
      • CPU л°”мҡҙл“ң мһ‘м—…: мҲңмҲҳн•ҳкІҢ CPU м§‘м•Ҫм Ғмқё лҸҷкё° мһ‘м—…мқҖ л‘җ нҷҳкІҪ лӘЁл‘җм—җм„ң мӢӨн–ү мҠӨл Ҳл“ңлҘј кі„мҶҚ м°ЁлӢЁн•©лӢҲлӢӨ.
